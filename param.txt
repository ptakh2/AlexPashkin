# Имя конечной программы (исполняемого файла)
TARGET = program
# Исходные файлы
SOURCES = program.c
# Компилятор
CC = gcc
# Флаги компиляции (предупреждения и стандарт C17)
CFLAGS = -Wall -Wextra -std=c17

# Правило по умолчанию (выполняется при команде `make`)
all: $(TARGET)

# Как собрать программу из исходников
$(TARGET): $(SOURCES)
	$(CC) $(CFLAGS) -o $(TARGET) $(SOURCES)

# Очистка (удаление собранных файлов)
clean:
	rm -f $(TARGET)

# Правило, которое не создаёт файл с именем clean (важно для make)
.PHONY: all clean
